# 安装

# NPM包 安装

  npm install holo-ui

  # or

  yarn add holo-ui

# CDN 安装

<!-- 引入样式 -->
<link rel="stylesheet" href="https://localhost:8080/holo-ui/lib/index/index.css">
<!-- 引入组件库 -->
<script src="https://localhost:8080/holo-ui/lib/index/index.js"></script>
